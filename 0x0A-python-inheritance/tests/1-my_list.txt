import sys

MyList = __import__('1-my_list').MyList

    """Test cases for MyList class"""

    def test_sorted_list(self):
        """Test that a list is printed sorted"""
        my_list = MyList()
        my_list.append(1)
        my_list.append(4)
        my_list.append(2)
        my_list.append(3)
        my_list.append(5)

        # check original list is not sorted
        self.assertNotEqual(my_list, [1, 2, 3, 4, 5])

        # check sorted method returns the sorted list
        self.assertEqual(sorted(my_list), [1, 2, 3, 4, 5])
